```bash
npm run e2e:build:ios       # Build E2E test app for iOS
npm run e2e:build:android   # Build E2E test app for Android
npm run e2e:test:ios        # Run E2E tests on iOS
npm run e2e:test:android    # Run E2E tests on Android

# Production
expo build:ios              # Build iOS app for production
expo build:android          # Build Android app for production
expo publish                # Publish to Expo
```
